// Функция красивого форматирования чисел (добавляет пробелы: 1 000 000 ₽) function formatCurrency(value) { return Math.round(value).toLocaleString('ru-RU') + " ₽"; } // Формула расчета аннуитетного (равного) ежемесячного платежа function calculateMonthlyPayment(principal, annualRate, years) { if (principal <= 0 || years <= 0) return 0; const monthlyRate = annualRate / 12 / 100; const totalMonths = years * 12; if (monthlyRate === 0) return principal / totalMonths; return principal * (monthlyRate * Math.pow(1 + monthlyRate, totalMonths)) / (Math.pow(1 + monthlyRate, totalMonths) - 1);
