import requests from bs4 import BeautifulSoup import pandas as pd import time import random from fake_useragent import UserAgent # Настройки HEADERS = {'User-Agent': UserAgent().random} BASE_URL = 'https://example.com' # Замени на свой URL OUTPUT_FILE = 'data.csv' def get_soup(url): """Получает HTML страницу""" try: response = requests.get(url, headers=HEADERS, timeout=10) response.raise_for_status() return BeautifulSoup(response.text, 'html.parser') except Exception as e: print(f"Ошибка при загрузке {url}: {e}") return None def parse_page(soup): """Парсит данные со страницы""" items = [] # Пример парсинга - замени на свои селекторы for element in soup.find_all('div', class_='item'): title = element.find('h2').text.strip() if element.find('h2') else '' price = element.find('span', class_='price').text.strip() if element.find('span', class_='price') else '' link = element.find('a')['href'] if element.find('a') else '' items.append({ 'title': title, 'price': price, 'link': BASE_URL + link if link else '' }) return items def main(): all_data = [] for page in range(1, 6): # Парсим 5 страниц print(f"Парсинг страницы {page}...") url = f"{BASE_URL}/page/{page}" soup = get_soup(url) if soup: page_data = parse_page(soup) all_data.extend(page_data) # Случайная задержка time.sleep(random.uniform(1, 3)) # Сохраняем в CSV if all_data: df = pd.DataFrame(all_data) df.to_csv(OUTPUT_FILE, index=False, encoding='utf-8-sig') print(f"Сохранено {len(all_data)} записей в {OUTPUT_FILE}") else: print("Данные не найдены") if __name__ == "__main__": main()
